model model.clear( [chain], [all=true] )

Метод model.clear удаляет всю модель или её часть, переданную цепочкой первым аргументом (chain).
Если передан второй аргумент (all), то будут удалены также все колбэки, созданные с помощью model.follow и model.composite.

Пример

var model = Follow('sample'); model({ title: 'Hello world', data: [{x: 0}, {x: 1}, {y: 2}] }); model.follow('data', function(data){ alert(data); }); // удалим цепочку "title", что аналогично в данном случае model('title', undefined); model.clear('title'); alert( model ); // изменяем значение для цепочки "data", так что сработает колбэк с алертом model('data', 'New value'); // очистим всю модель со всеми колбэками model.clear(false, true); alert( model ); // изменяем еще раз значение "data", но никаких колбэков это не влечет, т.к все они удалены выше model('data', [1,2,3]); alert( model ); model.backup
